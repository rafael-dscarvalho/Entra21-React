1- SELECT NOME_LIVRO FROM LIVROS WHERE CATEGORIA = 'Romance';

2- SELECT DISTINCT LIVROS.EDITORA FROM LIVROS JOIN ESTOQUE ON LIVROS.ID_LIVRO = ESTOQUE.ID_LIVRO WHERE QTD_ESTOQUE > 1

3- SELECT VENDAS.ID_PEDIDO, VENDEDORES.NOME_VENDEDOR FROM VENDAS JOIN VENDEDORES ON VENDAS.ID_VENDEDOR = VENDEDORES.ID_VENDEDOR;

4- SELECT LIVROS.NOME_LIVRO, ESTOQUE.QTD_ESTOQUE FROM LIVROS JOIN ESTOQUE ON LIVROS.ID_LIVRO = ESTOQUE.ID_LIVRO;

5- SELECT VENDEDORES.NOME_VENDEDOR FROM VENDEDORES LEFT JOIN VENDAS ON VENDEDORES.ID_VENDEDOR = VENDAS.ID_VENDEDOR WHERE VENDAS.ID_PEDIDO = NULL;

6- SELECT VENDAS.ID_PEDIDO, VENDEDORES.NOME_VENDEDOR FROM VENDAS JOIN VENDEDORES ON VENDAS.ID_VENDEDOR = VENDEDORES.ID_VENDEDOR;

7- SELECT LIVROS.CATEGORIA, COUNT(*), MAX(LIVROS.PRECO), MIN(LIVROS.PRECO), AVG(LIVROS.PRECO), SUM(ESTOQUE.QTD_ESTOQUE) FROM LIVROS LEFT JOIN ESTOQUE ON LIVROS.ID_LIVRO = ESTOQUE.ID_LIVRO GROUP BY LIVROS.CATEGORIA;

8- SELECT NOME_LIVRO, PRECO FROM LIVROS ORDER BY PRECO DESC;

9- SELECT NOME_LIVRO, PRECO FROM LIVROS ORDER BY PRECO DESC LIMIT 10;

10- SELECT VENDEDORES.NOME_VENDEDOR, SUM(VENDAS.QTD_VENDIDA) AS 'TOTAL_VENDAS' FROM VENDEDORES JOIN VENDAS ON VENDEDORES.ID_VENDEDOR = VENDAS.ID_VENDEDOR GROUP BY VENDEDORES.NOME_VENDEDOR HAVING SUM(QTD_VENDIDA) > 5;
